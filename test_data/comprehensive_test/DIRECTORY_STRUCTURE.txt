Quick Commands
==============

# Run all tests
cd /home/user/webapp/test_data/comprehensive_test
python3 run_all_tests.py

# Test individual scenario
./test_scenario.sh scenario_1_perfect_match

# View scenario details
cat scenario_1_perfect_match/client_info.json | python3 -m json.tool

# List all PDFs
find . -name "*.pdf" | sort

# Check PDF content
python3 -c "import PyPDF2; f=open('scenario_1_perfect_match/bank_statement_matching.pdf','rb'); print(PyPDF2.PdfReader(f).pages[0].extract_text())"
